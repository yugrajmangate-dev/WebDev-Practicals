Q09Library Signup — JS Form ValidationJavaScript
signup_validation.html
<!DOCTYPE html>
<html>
<head>
  <title>Library Signup</title>
  <style>
    body { font-family: Arial; padding: 20px; }
    input { display: block; margin: 5px 0 2px; padding: 7px; width: 260px; }
    .err { color: red; font-size: 12px; }
    button { margin-top: 12px; padding: 8px 20px; background: navy; color: white; border: none; cursor: pointer; }
  </style>
</head>
<body>
  <h2>Library Signup</h2>
  <form onsubmit="return validate()">
    Name:    <input type="text"   id="name">   <span class="err" id="e1"></span>
    Email:   <input type="text"   id="email">  <span class="err" id="e2"></span>
    Mobile:  <input type="text"   id="mob">    <span class="err" id="e3"></span>
    Password:<input type="password" id="pass"> <span class="err" id="e4"></span>
    <button>Sign Up</button>
  </form>

<script>
function validate() {
  let ok = true;
  // clear errors
  ['e1','e2','e3','e4'].forEach(id => document.getElementById(id).innerText = '');

  let name  = document.getElementById('name').value.trim();
  let email = document.getElementById('email').value.trim();
  let mob   = document.getElementById('mob').value.trim();
  let pass  = document.getElementById('pass').value;

  if (name == '') {
    document.getElementById('e1').innerText = 'Name required'; ok = false;
  }
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
    document.getElementById('e2').innerText = 'Invalid email'; ok = false;
  }
  if (!/^[6-9]\d{9}$/.test(mob)) {
    document.getElementById('e3').innerText = 'Invalid mobile (10 digits)'; ok = false;
  }
  if (pass.length < 6) {
    document.getElementById('e4').innerText = 'Min 6 characters'; ok = false;
  }

  if (ok) alert('Signup Successful!');
  return false;
}
</script>
</body>
</html>

  let name  = document.getElementById('name').value;
  let marks = document.getElementById('marks').value;

  // Create new element using DOM
  let p = document.createElement('p');
  p.innerHTML = count + '. <b>' + name + '</b> — Marks: ' + marks;

  // Change heading dynamically
  document.querySelector('h2').innerText = 'Students Added: ' + count;

  document.getElementById('output').innerHTML = '';
  document.querySelector('h2').innerText = 'DOM Dynamic Content';

Q11Simple Calculator — switch-caseJavaScript
calculator.html
<!DOCTYPE html>
<html>
<head><title>Calculator</title></head>
<body>
  <h2>Simple Calculator</h2>
  Number 1: <input type="number" id="n1"><br><br>
  Number 2: <input type="number" id="n2"><br><br>

  <button onclick="calc('+')'">Add</button>
  <button onclick="calc('-')">Subtract</button>
  <button onclick="calc('*')">Multiply</button>
  <button onclick="calc('/')">Divide</button>

  <p id="result"></p>

<script>
function calc(op) {
  let a = parseFloat(document.getElementById('n1').value);
  let b = parseFloat(document.getElementById('n2').value);
  let res;

  switch(op) {
    case '+': res = a + b; break;
    case '-': res = a - b; break;
    case '*': res = a * b; break;
    case '/': res = (b != 0) ? a / b : 'Cannot divide by zero'; break;
    default:  res = 'Invalid';
  }

  document.getElementById('result').innerHTML = '<b>Result: ' + res + '</b>';
}
</script>
</body>
</html>

Q16Area Calculator — Rectangle, Triangle, CircleJavaScript
area.html
<!DOCTYPE html>
<html>
<head><title>Area Calculator</title></head>
<body>
  <h2>Area Calculator</h2>
  Shape:
  <select id="shape" onchange="showFields()">
    <option value="">-- Select --</option>
    <option value="rect">Rectangle</option>
    <option value="tri">Triangle</option>
    <option value="circ">Circle</option>
  </select>
  <br><br>
  <div id="fields"></div>
  <p id="result"></p>

<script>
function showFields() {
  let s = document.getElementById('shape').value;
  let html = '';
  switch(s) {
    case 'rect': html = 'Length: <input id="l"> Width: <input id="w">'; break;
    case 'tri':  html = 'Base: <input id="b"> Height: <input id="h">';  break;
    case 'circ': html = 'Radius: <input id="r">';                        break;
  }
  if (html) html += ' <button onclick="calcArea()">Calculate</button>';
  document.getElementById('fields').innerHTML = html;
}

function calcArea() {
  let s = document.getElementById('shape').value;
  let area;
  switch(s) {
    case 'rect': area = document.getElementById('l').value * document.getElementById('w').value; break;
    case 'tri':  area = 0.5 * document.getElementById('b').value * document.getElementById('h').value; break;
    case 'circ': area = (Math.PI * Math.pow(document.getElementById('r').value, 2)).toFixed(2); break;
  }
  document.getElementById('result').innerHTML = '<b>Area = ' + area + ' sq.units</b>';
}
</script>
</body>
</html>

  let ok = true;
  ['e1','e2','e3','e4','e5'].forEach(id => document.getElementById(id).innerText = '');

  let name  = document.getElementById('name').value.trim();
  let email = document.getElementById('email').value.trim();

function calculate() {
  let name  = document.getElementById('sname').value;
  let marks = [
    +document.getElementById('m1').value,
    +document.getElementById('m2').value,
    +document.getElementById('m3').value,
    +document.getElementById('m4').value,
    +document.getElementById('m5').value
  ];

  let total   = marks.reduce((a, b) => a + b, 0);
  let percent = (total / 500) * 100;
  let grade;

  else if (percent >= 60) grade = 'B';

                  '<b>Total:</b> ' + total + ' / 500<br>' +
